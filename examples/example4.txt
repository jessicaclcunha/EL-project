start: Agenda

Agenda     -> DeclXML AAGENDA Lista FAGENDA

DeclXML    -> DCA ListaAtrib DCF

ListaAtrib -> Atrib ListaAtrib | epsilon

Atrib      -> ID '=' VATRIB

Lista      -> Elem Lista | epsilon

Elem       -> Entrada | Grupo

Entrada    -> AENTRADA ListaAtrib '>' Nome EntradaCont

EntradaCont -> Telefone FENTRADA
             | Email Telefone FENTRADA

Nome     -> ANOME STRING FNOME
Email    -> AEMAIL STRING FEMAIL
Telefone -> ATELEFONE STRING FTELEFONE

Grupo  -> AGRUPO ListaAtrib '>' GLista FGRUPO

GLista -> GElem GLista | epsilon

GElem  -> Entrada | Grupo | Ref

Ref    -> AREF ListaAtrib '/' '>'


AAGENDA   = /<agenda>/
FAGENDA   = /<\/agenda>/
DCA       = /<\?xml[^?]*/
DCF       = /\?>/
FENTRADA  = /<\/entrada>/
AENTRADA  = /<entrada/
FGRUPO    = /<\/grupo>/
AGRUPO    = /<grupo/
FNOME     = /<\/nome>/
ANOME     = /<nome>/
FEMAIL    = /<\/email>/
AEMAIL    = /<email>/
FTELEFONE = /<\/telefone>/
ATELEFONE = /<telefone>/
AREF      = /<ref/
VATRIB    = /"[^"]*"/
STRING    = /\d[^<>=]*|[_a-zA-Z]\w*[^<>"=\w][^<>=]*/
ID        = /[_a-zA-Z]\w*/
